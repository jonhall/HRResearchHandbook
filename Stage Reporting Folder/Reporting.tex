\usepackage{ifthen}

\newcommand{\ritem}[2]{%
	\ifthenelse{\equal{#1}{!}}
		%%new section
		{\item[\color{red}\textbullet]\color{black}}
		{\ifthenelse{\equal{#1}{*}}
			%%updated section
			{\item[\color{black}\textbullet]\color{black}}
			%%not yet included						
			{\item[\color{gray4}\textbullet]\color{gray4}}
		}
	#2}

\newcommand{\titem}[2]{%
	\ifthenelse{\equal{#1}{*}}%
		{\color{black}}%
		{\color{gray4}}%
	#2}

\newcommand\ReportTitle[1]{\reportheader \titem{#1}{Title}}

}
